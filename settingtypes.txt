# escalator mod – settingtypes.txt
# These settings can be adjusted in the Luanti main-menu settings panel
# under "Mods" → "escalator".

# Horizontal speed of entities on the escalator (nodes per second).
escalator_h_speed (Escalator horizontal speed) float 1.5 0.5 5.0

# Vertical speed component applied while on the escalator (nodes per second).
escalator_v_speed (Escalator vertical speed) float 1.5 0.5 5.0

# Maximum number of stair nodes scanned ahead of the controller.
escalator_max_stair_length (Escalator max stair scan length) int 32 4 128

# Node-timer interval for the controller (seconds). Lower = smoother but heavier on CPU.
escalator_timer_interval (Escalator timer interval) float 0.15 0.05 1.0

# Maximum number of controller blocks that may be stacked vertically.
escalator_max_stack (Escalator max stack height) int 10 1 10
